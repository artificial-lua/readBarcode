CREATE TABLE information
(
    `num`     INT            NOT NULL    AUTO_INCREMENT COMMENT 'User_Number', 
    `id`      VARCHAR(45)    NOT NULL    COMMENT 'User_ID', 
    `signup`  TIMESTAMP      NOT NULL    COMMENT 'Signup_date', 
     PRIMARY KEY (num)
);

ALTER TABLE information COMMENT 'User Information';

CREATE UNIQUE INDEX UQ_information_1
    ON information(id);


CREATE TABLE rawdata
(
    `id`        INT            NOT NULL    AUTO_INCREMENT COMMENT 'Barcode_ID', 
    `raw`       VARCHAR(45)    NULL        COMMENT 'Barcode_raw', 
    `user_num`  INT            NULL        COMMENT 'User_Number', 
    `time`      TIMESTAMP      NULL        COMMENT 'Date_Time', 
    `title`     VARCHAR(45)    NULL        COMMENT 'Barcode_title', 
    `des`       VARCHAR(45)    NULL        COMMENT 'Descript', 
     PRIMARY KEY (id)
);

ALTER TABLE rawdata COMMENT 'Barcode Rawdata';

ALTER TABLE rawdata
    ADD CONSTRAINT FK_rawdata_user_num_information_num FOREIGN KEY (user_num)
        REFERENCES information (num) ON DELETE RESTRICT ON UPDATE RESTRICT;


CREATE TABLE barcoderait
(
    `num`       INT         NOT NULL    AUTO_INCREMENT COMMENT 'Rait_Number', 
    `raw_id`    INT         NULL        COMMENT 'Raw_ID', 
    `user_num`  INT         NULL        COMMENT 'User_Number', 
    `rait`      BIT         NULL        COMMENT 'Raw_Rait', 
    `time`      DATETIME    NULL        COMMENT 'Date_Time', 
     PRIMARY KEY (num)
);

ALTER TABLE barcoderait COMMENT 'Barcode Rait';

ALTER TABLE barcoderait
    ADD CONSTRAINT FK_barcoderait_raw_id_rawdata_id FOREIGN KEY (raw_id)
        REFERENCES rawdata (id) ON DELETE RESTRICT ON UPDATE RESTRICT;

ALTER TABLE barcoderait
    ADD CONSTRAINT FK_barcoderait_user_num_information_num FOREIGN KEY (user_num)
        REFERENCES information (num) ON DELETE RESTRICT ON UPDATE RESTRICT;
